#### **Sharding Strategy**
```javascript
// Sharding por módulo
const getShard = (moduleId) => {
  const shards = ['shard1', 'shard2', 'shard3'];
  const hash = moduleId.split('').reduce((a, b) => {
    a = ((a << 5) - a) + b.charCodeAt(0);
    return a & a;
  }, 0);
  return shards[Math.abs(hash) % shards.length];
};
```
