            'compliance_scores': compliance_score,
            'total_score': total_score,
            'overall_compliance': self.get_compliance_level(total_score),
            'recommendations': self.generate_recommendations(compliance_score)
        }
    
    def evaluate_principle(self, principle, campaign):
        # Evaluar principio espec√≠fico
        if principle == 'lawfulness':
            return self.evaluate_lawfulness(campaign)
        elif principle == 'transparency':
            return self.evaluate_transparency(campaign)
        elif principle == 'data_minimization':
            return self.evaluate_data_minimization(campaign)
        # ... otros principios
        return 0.5  # Score por defecto

            'demographic_parity': 'Paridad demogr√°fica',
            'equalized_odds': 'Probabilidades igualadas',
            'calibration': 'Calibraci√≥n',
            'individual_fairness': 'Equidad individual',

    def detect_bias(self, model, dataset, protected_attributes):
        # Detectar sesgos en el modelo
        bias_report = {}

            bias_score = self.calculate_bias_score(model, dataset, protected_attributes, bias_type)
            bias_report[bias_type] = {
                'score': bias_score,
                'description': description,
                'bias_level': self.get_bias_level(bias_score),
                'recommendations': self.generate_bias_recommendations(bias_type, bias_score)
            }
        
        return bias_report
    
    def calculate_bias_score(self, model, dataset, protected_attributes, bias_type):
        # Calcular score de sesgo espec√≠fico
        if bias_type == 'demographic_parity':
            return self.calculate_demographic_parity(model, dataset, protected_attributes)
        elif bias_type == 'equalized_odds':
            return self.calculate_equalized_odds(model, dataset, protected_attributes)
        elif bias_type == 'calibration':
            return self.calculate_calibration(model, dataset, protected_attributes)
        # ... otros tipos de sesgo
        
        return 0.0
    
    def calculate_demographic_parity(self, model, dataset, protected_attributes):
        # Calcular paridad demogr√°fica
        predictions = model.predict(dataset)
        
        demographic_parity_scores = []
        for attr in protected_attributes:
            for value in dataset[attr].unique():
                group_mask = dataset[attr] == value
                group_positive_rate = predictions[group_mask].mean()
                demographic_parity_scores.append(group_positive_rate)
        
        # Calcular diferencia m√°xima entre grupos
        max_difference = max(demographic_parity_scores) - min(demographic_parity_scores)
        
        return max_difference
